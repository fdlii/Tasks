SELECT DISTINCT maker
FROM product
WHERE maker IN (
    SELECT DISTINCT maker
    FROM product
    JOIN printer ON product.model = printer.model
)
AND maker IN (
    SELECT p.maker
    FROM product p
    JOIN pc ON p.model = pc.model
    WHERE pc.ram = (SELECT MIN(ram) FROM pc)
      AND pc.speed = (
          SELECT MAX(speed)
          FROM pc
          WHERE ram = (SELECT MIN(ram) FROM pc)
      )
)
ORDER BY maker;
