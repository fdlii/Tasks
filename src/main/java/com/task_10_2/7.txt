SELECT product.model, price
FROM product
JOIN pc ON product.model = pc.model
WHERE product.maker = 'B'

UNION ALL

SELECT product.model, price
FROM product
JOIN laptop ON product.model = laptop.model
WHERE product.maker = 'B'

UNION ALL

SELECT product.model, price
FROM product
JOIN printer ON product.model = printer.model
WHERE product.maker = 'B'
